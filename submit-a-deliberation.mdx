<Steps>
  <Step title="From the Home page">
    1. Select the plus icon in the upper right corner
    2. Select if it is a small, medium, or big decision
    3. Enter the question you are trying to deliberate
    4. Enter the possible options
    5. Select "Deliberate" in the upper right corner
  </Step>
  <Step title="Completed!">
    You have now started a deliberation.
  </Step>
</Steps>
