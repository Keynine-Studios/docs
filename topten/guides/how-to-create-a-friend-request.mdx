<Steps>
  <Step title='From the "Explore" tab'>
    1. Click "Friends" on the bottom bar
    2. Click into the search bar
    3. Enter the friends username
    4. Click "Add"
    5. Click "OK"
  </Step>
  <Step title="Completed!">
    You have now created a friend request!
  </Step>
</Steps>
