<Steps>
  <Step title="From the main page">
    1. Click on "My Top 10's" on the bottom navigation bar
    2. Click on the trip you want to add a picture to
    3. Click on the category the photo goes in
    4. To upload a photo, click "Choose from Library", otherwise click "Take Photo"
    5. Add a caption (optional)
    6. Click "Save Photo"
  </Step>
  <Step title="Completed!">
    You have now uploaded a photo to Top 10!
  </Step>
</Steps>
