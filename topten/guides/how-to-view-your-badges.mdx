<Steps>
  <Step title='From the "Explore" Tab'>
    1. Click on "Profile" on the bottom bar
    2. Scroll down to the "Badges" section
    3. Wait for the badges to load
    4. Click on one to enlarge it
  </Step>
  <Step title="Completed!">
    You now can view your badges in Top 10!
  </Step>
</Steps>
