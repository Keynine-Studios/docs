<Steps>
  <Step title="From the Explore Tab">
    1. Click the "Friends" tab
    2. Click "Requests"
    3. To accept, click the check, otherwise click the "X"
  </Step>
  <Step title="Completed!">
    You have successfully responded to a friend request!
  </Step>
</Steps>
