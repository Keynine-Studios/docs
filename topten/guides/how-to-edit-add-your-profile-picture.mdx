<Steps>
  <Step title='From the "Explore" tab'>
    1. Click on "Profile"
    2. Scroll Down
    3. Click "Edit Profile"
    4. Scroll Up
    5. Click the camera icon on the profile picture
    6. Select either "Take or Upload a photo"
    7. Scroll Down
    8. Click "Save Changes"
  </Step>
  <Step title="Completed!">
    You have successfully added/edited a profile picture.
  </Step>
</Steps>
