</Warning>

<Steps>
  <Step title='From the "Explore" tab'>
    1. Click "Friends"
    2. Click on the friend you want to remove
    3. Click the "..." (ellipse) in the upper right corner
    4. Click "Remove Friend"
    5. Click "Remove"
  </Step>
  <Step title="Completed!">
    You have successfully removed a friend!
  </Step>
</Steps>
