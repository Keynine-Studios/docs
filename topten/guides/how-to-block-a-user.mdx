<Steps>
  <Step title='From the "Explore" tab'>
    1. Click "Friends" on the bottom bar
    2. Click the friend you want to block
    3. Click the "..." (ellipse) in the top right corner
    4. Click "Block User"
    5. Click "Block"
  </Step>
  <Step title="Completed!">
    You have successfully blocked the user!
  </Step>
</Steps>
