<Steps>
  <Step title='From the "Explore" Tab'>
    1. Click in the search box
    2. Enter the name of the city
    3. Click the "\+" button
    4. Confirm by clicking "Add"
  </Step>
  <Step title="Completed!">
    You have now added a new location on Top 10!
  </Step>
</Steps>
