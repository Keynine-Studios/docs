<Steps>
  <Step title='From the "Explore" tab'>
    1. Click "Profile" on the bottom bar
    2. Scroll Down
    3. Click "Edit Profile"
    4. Scroll Up
    5. Enter the new username you want in the text field
    6. Scroll Down
    7. Click "Save Changes"
  </Step>
  <Step title="Completed!">
    You have successfully changed your username!
  </Step>
</Steps>
