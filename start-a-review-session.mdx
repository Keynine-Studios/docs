<Steps>
  <Step title="From the Skills page">
    1. Select the "Review" page
    2. Select the skill you would like to review
    3. Select the units you would like to review
    4. You can choose if you want "Infinite Review" which makes the review continue endlessly.
    5. Select "Begin Session"
  </Step>
  <Step title="Completed!">
    You have now started a review session!
  </Step>
</Steps>
