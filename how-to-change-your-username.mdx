<Steps>
  <Step title="From the Skills page">
    1. Select the "Profile" page
    2. Select the pencil icon next to your username
    3. Enter a new username.
    4. Click "Save Username"
  </Step>
  <Step title="Completed!">
    You have now successfully updated your username!
  </Step>
</Steps>
